#version 460 core

layout (location = 0) in vec3 v_position;
layout (location = 1) in vec3 v_normal;
layout (location = 2) in vec2 v_tex_coords;
layout (location = 3) in vec3 v_tangent;
layout (location = 4) in vec3 v_bitangent;
layout (location = 5) in mat4 instanceMatrix;

out vec3 frag_pos;
out vec2 tex_coords;
out mat3 tbn;

uniform mat4 projectionMatrix;
uniform mat4 viewMatrix;

void main() 
{
	vec3 t = normalize(vec3(instanceMatrix * vec4(v_tangent, 0.0)));
	vec3 b = normalize(vec3(instanceMatrix * vec4(v_bitangent, 0.0)));
	vec3 n = normalize(vec3(instanceMatrix * vec4(v_normal, 0.0)));
	tbn = mat3(t, b, n);

	frag_pos = vec3(instanceMatrix * vec4(v_position, 1.0));
	tex_coords = v_tex_coords; 

	gl_Position = projectionMatrix * viewMatrix * vec4(frag_pos, 1.0);
};
